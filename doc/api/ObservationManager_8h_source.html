33class ObservationManager : public ManagerBase<ObservationTerm> {
34 public:
35 explicit ObservationManager(const size_t historySize = 1, bool byTerms = true)
36 : ManagerBase(), historyLength_(historySize), byTerms_(byTerms) {}
37
38 vector_t getValue() override {
39 vector_t value(getSize());
40 history_.pop_front();
41 history_.push_back(ManagerBase::getValue());
42
43 if (byTerms_) {
44 size_t accumulatedT = 0;
45 size_t accumulatedI = 0;
46 for (size_t t = 0; t < terms_.size(); ++t) {
47 const size_t termSize = terms_[t]->getSize();
48 for (size_t i = 0; i < historyLength_; ++i) {
49 value.segment(accumulatedI, termSize) = history_[i].segment(accumulatedT, termSize);
50 accumulatedI += termSize;
51 }
52 accumulatedT += termSize;
53 }
54 } else {
55 for (size_t i = 0; i < historyLength_; ++i) {
56 value.segment(i * ManagerBase::getSize(), ManagerBase::getSize()) = history_[i];
57 }
58 }
59 return value;
60 }
61
62 size_t getSize() const override {
63 const size_t signalStepSize = ManagerBase::getSize();
64 return historyLength_ * signalStepSize;
65 }
66
67 void reset() override {
68 ManagerBase::reset();
69 history_.clear();
70 for (size_t i = 0; i < historyLength_; ++i) {
71 history_.push_back(vector_t::Zero(getSize()));
72 }
73 }
74
75 protected:
76 size_t historyLength_;
77 bool byTerms_;
78 std::deque<vector_t> history_;
79};

166class PhaseObservationTerm final : public ObservationTerm {
167 public:
168 using ObservationTerm::ObservationTerm;
169 PhaseObservationTerm() : ObservationTerm(nullptr) {}
170 size_t getSize() const override { return 4; }
171 void reset() override {
172 phase_ = vector_t::Zero(2);
173 phase_[1] = M_PI;
174 }
175
176 protected:
177 vector_t evaluate() override {
178 phase_[0] = std::fmod(phase_[0] + phaseDt_ + M_PI, 2 * M_PI) - M_PI;
179 phase_[1] = std::fmod(phase_[1] + phaseDt_ + M_PI, 2 * M_PI) - M_PI;
180
181 vector_t obs(4);
182 vector_t phase(2);
183 phase = phase_;
184 obs[0] = std::cos(phase[0]);
185 obs[1] = std::cos(phase[1]);
186 obs[2] = std::sin(phase[0]);
187 obs[3] = std::sin(phase[1]);
188 return obs;
189 }
190
191 vector_t phase_;
192 scalar_t phaseDt_{2 * M_PI * 0.02 * 1.5};
193};
