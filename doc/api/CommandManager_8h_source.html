38class VelocityTopicCommandTerm : public CommandTerm {
39 public:
40 using Twist = geometry_msgs::msg::TwistStamped;
41 VelocityTopicCommandTerm(rclcpp_lifecycle::LifecycleNode::SharedPtr node, const std::string& topicName) : node_(std::move(node)) {
42 subscriber_ = node_->create_subscription<Twist>(topicName, 10, [this](const Twist::SharedPtr msg) { receivedMsg_.set(msg); });
43 }
44
45 vector_t getValue() override {
46 std::shared_ptr<Twist> lastCommandMsg;
47 receivedMsg_.get(lastCommandMsg);
48 const auto timeNow = node_->get_clock()->now();
49 if (lastCommandMsg->header.stamp.sec == 0 && lastCommandMsg->header.stamp.nanosec == 0) {
50 RCLCPP_WARN_ONCE(node_->get_logger(),
51 "Received TwistStamped with zero timestamp, setting it to current "
52 "time, this message will only be shown once");
53 lastCommandMsg->header.stamp = timeNow;
54 }
55 vector_t command = vector_t::Zero(3);
56 if (timeNow - lastCommandMsg->header.stamp < std::chrono::milliseconds{static_cast<int>(0.5 * 1000.0)}) {
57 command << lastCommandMsg->twist.linear.x, lastCommandMsg->twist.linear.y, lastCommandMsg->twist.angular.z;
58 }
59 return command;
60 }
61
62 void reset() override {
63 const auto twist = std::make_shared<Twist>();
64 receivedMsg_.set(twist);
65 }
66
67 size_t getSize() const override { return 3; }
68
69 protected:
70 rclcpp_lifecycle::LifecycleNode::SharedPtr node_;
71 rclcpp::Subscription<Twist>::SharedPtr subscriber_;
72 realtime_tools::RealtimeBox<std::shared_ptr<Twist>> receivedMsg_;
73};
