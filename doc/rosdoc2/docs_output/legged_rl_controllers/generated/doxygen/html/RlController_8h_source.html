135template <typename Functor1, typename Functor2>
136void executeAtRate(Functor1 f, Functor2 condition, double frequency) {
137 using clock = std::chrono::high_resolution_clock;
138
139 // Compute desired duration rounded to clock decimation
140 const std::chrono::duration<double> desiredDuration(1.0 / frequency);
141 const auto dt = std::chrono::duration_cast<clock::duration>(desiredDuration);
142
143 // Initialize timing
144 const auto start = clock::now();
145 auto sleepTill = start + dt;
146
147 // Execution loop
148 while (condition()) {
149 // Execute wrapped function
150 f();
151
152 // Sleep
153 std::this_thread::sleep_until(sleepTill);
154 sleepTill += dt;
155 }
156}
